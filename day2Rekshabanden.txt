int rakshaBandhan(vector < int > arr, int n) {

   

   sort(arr.begin(), arr.end(), greater<int>());

 

   long long sum = 0;

   int cnt = 0;

   for(int i = 0; i<n; i++) {

       sum += arr[i];          

       if (sum <= 0) {

           break;

       }

       cnt++;

   }

   return cnt;

}
